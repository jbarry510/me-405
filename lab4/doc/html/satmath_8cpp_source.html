27 /** \brief This function performs saturated addition
28  * \details The function performs addition of two 16-bit signed integers, saturating at either the maximal
29  * or minimal 16-bit values
30  * @param x Augend
31  * @param y Addend
32  * @return Sum
33  */
34 int16_t satmath::signed_saturated_add(int16_t x, int16_t y)
35 {
36  // Determine the lower or upper bound of the result
37  int16_t ret = (x < 0) ? INT16_MIN : INT16_MAX;
38  // this is always well defined:
39  // if x < 0 this adds a positive value to INT16_MIN
40  // if x > 0 this subtracts a positive value from INT16_MAX
41  int16_t comp = ret - x;
42  // the condition is equivalent to
43  // ((x < 0) && (y > comp)) || ((x >=0) && (y <= comp))
44  if ((x < 0) == (y > comp)) ret = x + y;
45  return ret;
46 }

49 /** \brief This function performs saturated subtraction
50  * \details The function performs subtraction of two 16-bit signed integers, saturating at either the
51  * maximal or minimal 16-bit values
52  * @param x Minuend
53  * @param y Subtrahend
54  * @return Difference
55  */
56 int16_t satmath::signed_saturated_sub(int16_t x, int16_t y)
57 {
58  // if they are equal return 0
59  if(x == y) return 0;
60 
61  // negate y
62  if (y == INT16_MIN && x < 0) x++;
63  y = (y == INT16_MIN) ? INT16_MAX:-y;
64  return signed_saturated_add(x,y);
65 }
