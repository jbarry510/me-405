266 /** \brief Computes the new PID output value
267  * \details This method forces the controller to recompute without updating the input or setpoint
268  */
269 int16_t pid::compute()
270 {
271  // Create a temp variable to hold control signal before output
272  int16_t temp = 0;
273 
274  // Compute Error
275  error = setpoint - input;
276 
277  // Integrate the error and subtract the current saturation value multiplied by the anti-windup gain.
278  esum = sssub(ssadd(esum,error), ssdiv(ssmul(config.Kw,saturation),1024));
279 
280  // Compute output based on current PID mode
281  switch(config.mode)
282  {
283  // No output signal while the controller is off
284  case OFF:
285  temp = 0;
286  break;
287 
288  // Proportional only
289  case P:
290  temp = ssdiv(ssmul(config.Kp,error),1024);
291  break;
292 
293  // Proportional and integral
294  case PI:
295  temp = ssadd(ssdiv(ssmul(config.Ki,esum),1024), ssdiv(ssmul(config.Kp,error),1024));
296  break;
297  // Proportional and Derivative
298  case PD:
299  temp = ssadd(ssdiv(ssmul(config.Kd,dinput),1024),ssdiv(ssmul(config.Kp,error),1024));
300  break;
301 
302  // Full PID
303  case PID:
304  temp = ssadd(ssdiv(ssmul(config.Ki,esum),1024),
305  ssadd(ssdiv(ssmul(config.Kd,dinput),1024),
306  ssdiv(ssmul(config.Kp,error),1024)));
307  break;
308 
309  // Manual mode (keep the current control value)
310  case MANUAL:
311  temp = output;
312  break;
313 
314  }
315 
316  // Saturation and storage of temp before exit
317  // During the saturation, the amount "saturated" is stored to use for anti-windup feedback
318  if (temp > config.max)
319  {
320  saturation = sssub(temp, config.max);
321  output = config.max;
322  }
323  else if (temp < config.min)
324  {
325  saturation = sssub(temp, config.min);
326  output = config.min;
327  }
328  else
329  {
330  saturation = 0;
331  output = temp;
332  }
333 
334  return output;
335 }
