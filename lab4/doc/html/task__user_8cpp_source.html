143  number_entered = 0; // Clears number_entered
144  sh_motor_select -> put(0); // Clears shared sh_motor_select
145  number_state = 0; // Clears number state
146  transition_to (1);
147  break;
148 
149  // A Ctrl-C character causes the CPU to restart
150  case (3):
151  *p_serial << PMS ("Resetting AVR") << endl;
152  wdt_enable (WDTO_120MS);
153  for (;;);

165  }; // End switch for characters

172  case (1):

176

178  if (char_in >= '0' && char_in <= '9')

181  number_entered *= 10;
182  number_entered += char_in - '0';
183  char_num_count++;
184  }
185 
186  // Dash, indicating a negative number
187  else if (char_in == ASCII_DASH)

191  }
192

197  *p_serial << (char)ASCII_SPACE; // Print Space
198  *p_serial << char_in; // Print Backspace
199 
200  number_entered /= 10; // Divide by 10 to remove ones place
201  char_num_count--; // Decrement until equal to zero
202  }
203 
204  // Line Feed, Carriage return is ignored; the newline character ends the entry
205  else if (char_in == ASCII_NEWLINE)
206  *p_serial << "\r";
207

209  else if (char_in == ASCII_RETURN)
