18 /** \brief This constructor sets up the motor object.
19  * \details The timer for the PWM signal is set up in fast PWM 8-bit mode with a clock
20  * prescaler of 8 which gives an operating frequency of approximately 8kHz. The

22  * motor based on the select variable. A debug message is printed stating if
23  * the motor constructor was sucessful or not.

28 motor_drv::motor_drv(emstream* p_serial_port, uint8_t motor_select)
29 {
30  // Defines pointer for serial to inputted parameter
31  ptr_to_serial = p_serial_port;
32  select = motor_select;
33 
34  // Timing channel 1 setup (Fast PWM, Clock prescaler of 8)
35  TCCR1A |= (1 << WGM10) | (1 << COM1B1) | (1 << COM1A1);
36  TCCR1A &= ~(1 << COM1B0) | ~(1 << COM1A0);
37  TCCR1B |= (1 << WGM12) | (1 << CS11);
38  TCCR1B &= ~(1 << CS12) | ~(1 << CS10);
39 
40  // If object is motor 1 then this block is used
41  if(select == 1)
42  {
43  // Sets direction of port C pins 0-2 to output (Direction control)
44  DDRC |= (1<<DDC0)|(1<<DDC1)|(1<<DDC2);
45 
46  // Sets output enable on motor driver 1
47  PINC |= (1 << PINC2);
48 
49  // Sets direction of port B pin 6 to output (PWM control)
50  DDRB |= (1<<DDB6);
51 
52  // Prints a message if motor 1 construction was sucessful
53  DBG(ptr_to_serial, "motor 1 constructor OK" << endl);
54  }
55 
56  // If object is motor 2 then this block is used
57  else if(select == 2)
58  {
59  // Sets direction of port D pins 5-7 to output (Direction control)
60  DDRD |= (1<<DDD5)|(1<<DDD6)|(1<<DDD7);
61 
62  // Sets output enable on motor driver 2
63  PIND |= (1 << PIND7);
64 
65  // Sets direction of port B pin 5 to output (PWM control)
66  DDRB |= (1<<DDB5);
67

91 void motor_drv::set_power(int16_t power)
92 {
93  // If object is motor 1 then this block is used
94  if(select == 1)
95  {
96  // Sets motor to turn clockwise and multiples power by -1 to get magnitude
97  if(power < 0)
98  {
99  PORTC &= ~(1<<PORTC0) | ~(1<<PORTC1); // Clears INA and INB of motor 1
100  PORTC |= (1<<PORTC1); // Sets INB of motor 1
101  PORTC &= ~(1<<PORTC0); // Clears INA of motor 1
102  power = power * -1;
103  }
104 
105  // Sets motor to turn counterclockwise
106  else
107  {
108  PORTC &= ~(1<<PORTC0) | ~(1<<PORTC1); // Clears INA and INB of motor 1
109  PORTC &= ~(1<<PORTC1); // Clears INB of motor 1
110  PORTC |= (1<<PORTC0); // Sets INA of motor 1
111  }
112 
113  // Sets output compare register for motor 1 PWM equal to the power
114  OCR1B = power;
115  }
116 
117  // If object is motor 2 then this block is used
118  if(select == 2)

120  // Sets motor to turn clockwise and multiples power by -1 to get magnitude

123  PORTD &= ~(1<<PORTD5) | ~(1<<PORTD6); // Clears INA and INB of motor 2
124  PORTD |= (1<<PORTD6); // Sets INB of motor 2

126  power = power*-1;
