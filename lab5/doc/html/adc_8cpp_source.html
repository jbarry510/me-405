65 /** @brief This method takes one A/D reading from the given channel and returns it.
66  * @details Forces the inputted channel to be a number from 0 to 7 then sets the
67  * ADMUX register to the correct channel. Next it starts the conversion and waits until
68  * it is finished. Finally it stores the result and returns it.
69  * @param ch The A/D channel which is being read must be from 0 to 7
70  * @return The result of the A/D conversion
71  */
72 
73 uint16_t adc::read_once (uint8_t ch)
74 {
75  // Initalizes result variable for A/D conversion
76  uint16_t result = 0;
77 
78  // Clears left 5 bits of channel
79  ch &= 0b00000111;
80 
81  // Clears right 3 bits of ADMUX and sets A/D channel
82  ADMUX &= 0b11111000;
83  ADMUX |= ch;
84 
85  // Starts conversion and waits until conversion is complete
86  ADCSRA |= 1<<ADSC;
87  while(ADCSRA & 0b01000000);
88 
89  // Stores the result of the conversion
90  result = ADC;
91 
92  return result;
93 }

106 uint16_t adc::read_oversampled (uint8_t channel, uint8_t samples)
107 {
108  // Intializes result and temporary samples variable
109  uint16_t result = 0;
110  uint8_t temp_samples = 0;
111 
112  // Checks if number of samples exceeds sample cap (assuming max A/D reading)
113  // Also saves number of samples in a temporary variable
114  if(samples >= 64)
115  {
116  samples = 60;
117  }
118  temp_samples = samples;
119 
120  // Cumulative sum of A/D readings to be averaged
121  for(int i = 0; i < samples; samples--)
122  {
123  result += read_once(channel);
124  }
125 
126  // Returns average of A/D readings
127  return result/temp_samples;
128  }
