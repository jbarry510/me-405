113  /** @brief Cause a stop condition on the I2C bus.
114  * @details This method causes a stop condition on the I2C bus. It's inline

116  * in assembly).
117  * @return @c false meaning OK because there's no test to show an error

119  bool stop (void)

152  /** @brief Take the mutex associated with this I2C bus.
153  * @details This method takes the mutex which controls access to this I2C bus.
154  * The mutex is automatically taken by the @c read() and @c write()
155  * commands, but when a device driver needs to use the @c read_byte()
156  * and @c write_byte() commands directly, that driver needs to handle
157  * the mutex with this command and the @c give_mutex() command.
158  */
159  void take_mutex (void)
160  {
161  xSemaphoreTake (mutex, portMAX_DELAY);
162  }
163 
164  /** @brief Give back the mutex associated with this I2C bus.
165  * @details This method gives the mutex which controls access to this I2C bus.
166  * It's a complement for the @c take_mutex() method.
167  */
168  void give_mutex (void)
169  {
170  xSemaphoreGive (mutex);
171  }
